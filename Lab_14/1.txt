#include <iostream>
#include <ctime>
#include <cstdlib>
#include <iomanip>
using namespace std;

int main() {

    srand(time(0));

    float *arr = new float[10];

    int min = -2;
    int max = 2;
    
    int positiveCount=0;
    int negativeCount=0;

    cout << "arr=" << endl;

    for(int i = 0; i < 10; i++) {

        arr[i] = min + (max - min) * (rand()%100)/100.0;

        cout << fixed << setprecision(2) << arr[i] << endl;
        
         if(arr[i] > 0) {
            positiveCount++;
        }
        else if(arr[i] < 0) {
            negativeCount++;
        }
    }

    cout << "\nPositive elements = " << positiveCount << endl;
    cout << "Negative elements = " << negativeCount << endl;
    

    delete [] arr;
    return 0;
}
