#include <iostream>
using namespace std;

bool isPrime(int n);

int main() {
    int number;

    cout << "Enter a number: ";
    cin >> number;

    if (isPrime(number)) {
        cout << "The number is prime";
    } else {
        cout << "The number is NOT prime";
    }

    return 0;
}
bool isPrime(int n) {
    
    if(n<=1){
        return false;
        }
    
    for(int i=2; i< n; i++){
        if(n % i==0)
            return false;
            
        }
         return true;
    }
